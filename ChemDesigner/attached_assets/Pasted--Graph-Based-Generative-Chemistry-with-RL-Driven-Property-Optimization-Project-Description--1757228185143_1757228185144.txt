# Graph-Based Generative Chemistry with RL-Driven Property Optimization

## Project Description

This project implements an AI-powered molecular generation system that combines **Graph Variational Autoencoders (GVAE)** with **Reinforcement Learning (RL)** to generate novel chemical compounds with desired properties. The system uses graph neural networks to understand molecular structures and reinforcement learning to optimize for specific pharmaceutical properties like drug-likeness (QED), synthetic accessibility (SA), and solubility.

### Key Features:
- **Graph-based molecular representation** using PyTorch Geometric
- **VAE-based generation** for creating novel molecular structures
- **RL optimization** using PPO to bias generation toward desired properties
- **Interactive web interface** built with React/TypeScript and Flask
- **Real-time property calculation** using RDKit
- **Modern full-stack architecture** with Flask backend and React frontend

### Target Users:
- Pharmaceutical companies for drug discovery
- Academic researchers in computational chemistry
- Chemistry students for educational purposes
- Biotechnology startups exploring AI-driven molecular design

**backend/models/graph_vae.py** - Graph VAE Model
```python
import torch
import torch.nn as nn
import torch.nn.functional as F
from torch_geometric.nn import GCNConv, global_mean_pool
from torch_geometric.data import Data, Batch

class GraphEncoder(nn.Module):
    """Graph encoder using GCN layers"""
    def __init__(self, input_dim, hidden_dim, latent_dim):
        super().__init__()
        self.conv1 = GCNConv(input_dim, hidden_dim)
        self.conv2 = GCNConv(hidden_dim, hidden_dim)
        self.conv3 = GCNConv(hidden_dim, hidden_dim)
        
        self.mean_layer = nn.Linear(hidden_dim, latent_dim)
        self.logvar_layer = nn.Linear(hidden_dim, latent_dim)
        
    def forward(self, x, edge_index, batch):
        # Graph convolutions
        h = F.relu(self.conv1(x, edge_index))
        h = F.relu(self.conv2(h, edge_index))
        h = F.relu(self.conv3(h, edge_index))
        
        # Global pooling
        h_graph = global_mean_pool(h, batch)
        
        # Output mean and log variance
        mu = self.mean_layer(h_graph)
        logvar = self.logvar_layer(h_graph)
        
        return mu, logvar

class GraphDecoder(nn.Module):
    """Graph decoder for molecule generation"""
    def __init__(self, latent_dim, hidden_dim, max_atoms=50, atom_types=9):
        super().__init__()
        self.latent_dim = latent_dim
        self.max_atoms = max_atoms
        self.atom_types = atom_types
        
        # Node generation
        self.node_layers = nn.Sequential(
            nn.Linear(latent_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, max_atoms * atom_types)
        )
        
        # Edge generation
        self.edge_layers = nn.Sequential(
            nn.Linear(latent_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, max_atoms * max_atoms * 4)  # 4 bond types
        )
        
    def forward(self, z):
        batch_size = z.size(0)
        
        # Generate node features
        node_logits = self.node_layers(z)
        node_logits = node_logits.view(batch_size, self.max_atoms, self.atom_types)
        
        # Generate adjacency matrix
        edge_logits = self.edge_layers(z)
        edge_logits = edge_logits.view(batch_size, self.max_atoms, self.max_atoms, 4)
        
        return node_logits, edge_logits

class GraphVAE(nn.Module):
    """Complete Graph Variational Autoencoder"""
    def __init__(self, input_dim=3, hidden_dim=128, latent_dim=64, max_atoms=50):
        super().__init__()
        self.latent_dim = latent_dim
        self.max_atoms = max_atoms
        
        self.encoder = GraphEncoder(input_dim, hidden_dim, latent_dim)
        self.decoder = GraphDecoder(latent_dim, hidden_dim, max_atoms)
        
    def reparameterize(self, mu, logvar):
        """Reparameterization trick"""
        std = torch.exp(0.5 * logvar)
        eps = torch.randn_like(std)
        return mu + eps * std
    
    def forward(self, data):
        # Encode
        mu, logvar = self.encoder(data.x, data.edge_index, data.batch)
        
        # Reparameterize
        z = self.reparameterize(mu, logvar)
        
        # Decode
        node_logits, edge_logits = self.decoder(z)
        
        return node_logits, edge_logits, mu, logvar
    
    def generate(self, z=None, num_samples=1):
        """Generate molecules from latent vectors"""
        if z is None:
            z = torch.randn(num_samples, self.latent_dim)
        elif isinstance(z, list):
            z = torch.tensor([z], dtype=torch.float32)
        
        with torch.no_grad():
            node_logits, edge_logits = self.decoder(z)
            
        return node_logits, edge_logits

def vae_loss(node_recon, edge_recon, node_target, edge_target, mu, logvar, beta=1.0):
    """VAE loss function"""
    # Reconstruction loss
    node_loss = F.cross_entropy(node_recon.view(-1, node_recon.size(-1)), 
                               node_target.view(-1), reduction='mean')
    edge_loss = F.cross_entropy(edge_recon.view(-1, edge_recon.size(-1)), 
                               edge_target.view(-1), reduction='mean')
    recon_loss = node_loss + edge_loss
    
    # KL divergence
    kl_loss = -0.5 * torch.mean(1 + logvar - mu.pow(2) - logvar.exp())
    
    return recon_loss + beta * kl_loss
```

```python
    """Generate molecules from latent space"""
    if models["vae"] is None:
        return {"smiles": ["CCO"] * n, "error": "Model not loaded"}
    
    try:
        vae = models["vae"]
        
        if z is not None:
            if isinstance(z, list):
                z = torch.tensor([z], dtype=torch.float32)
            n = z.shape[0]
        
        # Generate graphs
        node_logits, edge_logits = vae.generate(z=z, num_samples=n)
        
        # Convert to SMILES (simplified for demo)

            try:
                smi = graph_to_smiles(node_logits[i], edge_logits[i])

        return {"smiles": smiles}
    except Exception as e:
        return {"smiles": ["CCO"] * n, "error": str(e)}
```

```python
from rdkit import Chem

def calculate_properties(smiles: str):
    """Calculate molecular properties from SMILES"""
    mol = Chem.MolFromSmiles(smiles)
    if not mol:
        return {
            "smiles": smiles,
            "valid": False,
            "qed": 0.0,
            "logp": 0.0,
            "sa": 10.0,
            "mw": 0.0
        }
    
    try:
        qed = QED.qed(mol)
        logp = Descriptors.MolLogP(mol)
        mw = Descriptors.MolWt(mol)
        
        # Synthetic Accessibility Score (simplified)
        sa = rdMolDescriptors.CalcNumRings(mol) + rdMolDescriptors.CalcNumHBA(mol)
        
        return {
            "smiles": smiles,
            "valid": True,
            "qed": round(qed, 3),
            "logp": round(logp, 3),
            "sa": round(sa, 3),
            "mw": round(mw, 3)
        }
    except Exception as e:
        return {
            "smiles": smiles,
            "valid": False,
            "error": str(e),
            "qed": 0.0,
            "logp": 0.0,
            "sa": 10.0,

def smiles_to_graph(smiles):
    """Convert SMILES to graph representation"""
    mol = Chem.MolFromSmiles(smiles)
    if not mol:
        return None
    
    # Atom features
    atom_features = []
    for atom in mol.GetAtoms():
        features = [
            atom.GetAtomicNum(),
            int(atom.GetIsAromatic()),
            atom.GetFormalCharge()
        ]
        atom_features.append(features)
    
    # Bond indices and features
    edge_index = []
    edge_attr = []
    for bond in mol.GetBonds():
        i, j = bond.GetBeginAtomIdx(), bond.GetEndAtomIdx()
        edge_index.extend([[i, j], [j, i]])
        
        bond_type = bond.GetBondTypeAsDouble()
        edge_attr.extend([[bond_type], [bond_type]])
    
    # Convert to tensors
    x = torch.tensor(atom_features, dtype=torch.float)
    edge_index = torch.tensor(edge_index, dtype=torch.long).t().contiguous() if edge_index else torch.empty((2, 0), dtype=torch.long)
    edge_attr = torch.tensor(edge_attr, dtype=torch.float) if edge_attr else torch.empty((0, 1), dtype=torch.float)
    
    return Data(x=x, edge_index=edge_index, edge_attr=edge_attr)
```

**backend/services/rl_env.py** - RL Environment
```python
import gymnasium as gym
from gymnasium import spaces
import numpy as np
from services.props import calculate_properties, graph_to_smiles

class MolecularEnv(gym.Env):
    """Custom environment for molecular optimization"""
    
    def __init__(self, generator_model, latent_dim=64):
        super().__init__()
        self.generator = generator_model
        self.latent_dim = latent_dim
        
        # Action space: continuous latent vector
        self.action_space = spaces.Box(
            low=-3.0, high=3.0, shape=(latent_dim,), dtype=np.float32
        )
        
        # Observation space: current latent vector
        self.observation_space = spaces.Box(
            low=-np.inf, high=np.inf, shape=(latent_dim,), dtype=np.float32
        )
        
        self.current_latent = None
        
    def reset(self, seed=None, options=None):
        """Reset environment"""
        super().reset(seed=seed)
        self.current_latent = np.random.randn(self.latent_dim).astype(np.float32)
        return self.current_latent, {}
    
    def step(self, action):
        """Take a step in the environment"""
        # Update latent vector
        self.current_latent = np.clip(
            self.current_latent + action * 0.1, -3.0, 3.0
        ).astype(np.float32)
        
        # Generate molecule
        try:
            node_logits, edge_logits = self.generator.generate(
                z=torch.tensor([self.current_latent])
            )
            smiles = graph_to_smiles(node_logits[0], edge_logits[0])
            
            # Calculate reward
            props = calculate_properties(smiles)
            reward = self._calculate_reward(props)
            
        except Exception:
            reward = -1.0
            smiles = "INVALID"
        
        # Episode is always done for simplicity
        terminated = True
        truncated = False
        
        info = {"smiles": smiles, "properties": props if 'props' in locals() else {}}
        
        return self.current_latent, reward, terminated, truncated, info
    
    def _calculate_reward(self, props):
        """Calculate reward from molecular properties"""
        if not props.get("valid", False):
            return -1.0
        
        # Multi-objective reward
        qed_reward = props.get("qed", 0.0)
        sa_penalty = max(0, props.get("sa", 0) - 6) * 0.1  # Penalize high SA
        
        return qed_reward - sa_penalty
```

**scripts/data_preparation.py** - Data Processing
```python
import pandas as pd
from rdkit import Chem
import torch
from torch_geometric.data import Data
import tqdm
import os

def mol_to_graph(mol):
    """Convert RDKit molecule to PyTorch Geometric graph"""
    if mol is None:
        return None
    
    # Atom features
    atom_features = []
    for atom in mol.GetAtoms():
        features = [
            atom.GetAtomicNum(),
            int(atom.GetIsAromatic()),
            atom.GetFormalCharge()
        ]
        atom_features.append(features)
    
    # Bond indices and features
    edge_index = []
    edge_attr = []
    
    for bond in mol.GetBonds():
        i, j = bond.GetBeginAtomIdx(), bond.GetEndAtomIdx()
        edge_index.extend([[i, j], [j, i]])
        
        bond_type = bond.GetBondTypeAsDouble()
        edge_attr.extend([[bond_type], [bond_type]])
    
    # Convert to tensors
    x = torch.tensor(atom_features, dtype=torch.float)
    edge_index = torch.tensor(edge_index, dtype=torch.long).t().contiguous() if edge_index else torch.empty((2, 0), dtype=torch.long)
    edge_attr = torch.tensor(edge_attr, dtype=torch.float) if edge_attr else torch.empty((0, 1), dtype=torch.float)
    
    return Data(x=x, edge_index=edge_index, edge_attr=edge_attr)

def process_smiles_dataset(input_path, output_path, limit=10000, max_atoms=50):
    """Process SMILES dataset into graph format"""
    print(f"Processing SMILES from {input_path}")
    
    # Read SMILES data
    df = pd.read_csv(input_path)
    if limit:
        df = df.head(limit)
    
    print(f"Processing {len(df)} molecules...")
    
    graphs = []
    valid_count = 0
    
    for idx, row in tqdm.tqdm(df.iterrows(), total=len(df)):
        smiles = row['smiles']
        mol = Chem.MolFromSmiles(smiles)
        
        if mol and mol.GetNumAtoms() <= max_atoms:
            graph = mol_to_graph(mol)
            if graph is not None:
                graphs.append(graph)
                valid_count += 1
    
    print(f"Successfully processed {valid_count} valid molecules")
    
    # Save processed graphs
    torch.save(graphs, output_path)
    print(f"Saved processed graphs to {output_path}")
    
    return graphs

def create_train_val_split(graphs, val_ratio=0.2):
    """Split graphs into training and validation sets"""
    n_val = int(len(graphs) * val_ratio)
    indices = torch.randperm(len(graphs))
    
    train_graphs = [graphs[i] for i in indices[n_val:]]
    val_graphs = [graphs[i] for i in indices[:n_val]]
    
    return train_graphs, val_graphs

if __name__ == "__main__":
    # Paths
    input_file = "backend/data/zinc_smiles.csv"
    output_file = "backend/data/processed.pt"
    
    # Process dataset
    graphs = process_smiles_dataset(input_file, output_file, limit=5000)
    
    # Create train/val split

    # Save splits
    torch.save(train_graphs, "backend/data/train_graphs.pt")
    torch.save(val_graphs, "backend/data/val_graphs.pt")
    
    print(f"Training set: {len(train_graphs)} molecules")
    print(f"Validation set: {len(val_graphs)} molecules")
```

**scripts/train.py** - Training Script
```python
import torch
import torch.nn as nn
from torch_geometric.loader import DataLoader
from torch.optim import Adam
import os
import sys

# Add backend to path
sys.path.append(os.path.join(os.path.dirname(__file__), '..', 'backend'))

from models.graph_vae import GraphVAE, vae_loss
from services.rl_env import MolecularEnv
from stable_baselines3 import PPO

def train_vae(train_graphs, val_graphs, epochs=100, batch_size=32, lr=1e-3):
    """Train Graph VAE"""
    print("Training Graph VAE...")
    
    # Data loaders
    train_loader = DataLoader(train_graphs, batch_size=batch_size, shuffle=True)
    val_loader = DataLoader(val_graphs, batch_size=batch_size, shuffle=False)
    
    # Model
    model = GraphVAE(input_dim=3, hidden_dim=128, latent_dim=64)
    optimizer = Adam(model.parameters(), lr=lr)
    
    # Training loop
    for epoch in range(epochs):
        model.train()
        train_loss = 0
        
        for batch in train_loader:
            optimizer.zero_grad()
            
            # Forward pass
            node_recon, edge_recon, mu, logvar = model(batch)
            
            # Calculate loss (simplified)
            loss = vae_loss(node_recon, edge_recon, batch.x, batch.edge_index, mu, logvar)
            
            # Backward pass
            loss.backward()
            optimizer.step()
            
            train_loss += loss.item()
        
        # Validation
        model.eval()
        val_loss = 0
        with torch.no_grad():
            for batch in val_loader:
                node_recon, edge_recon, mu, logvar = model(batch)
                loss = vae_loss(node_recon, edge_recon, batch.x, batch.edge_index, mu, logvar)
                val_loss += loss.item()
        
        print(f"Epoch {epoch+1}/{epochs}: Train Loss: {train_loss/len(train_loader):.4f}, Val Loss: {val_loss/len(val_loader):.4f}")
    
    # Save model
    torch.save(model.state_dict(), "backend/models/gvae_trained.pth")
    print("Saved trained VAE model")
    
    return model

def train_rl(vae_model, total_timesteps=100000):
    """Train RL agent for property optimization"""
    print("Training RL agent...")
    
    # Create environment
    env = MolecularEnv(vae_model)
    
    # Create PPO agent
    model = PPO("MlpPolicy", env, verbose=1, learning_rate=3e-4)
    
    # Train
    model.learn(total_timesteps=total_timesteps)
    
    # Save model
    model.save("backend/models/ppo_molgen")
    print("Saved trained RL model")
    
    return model

def main():
    # Load data
    print("Loading processed data...")
    train_graphs = torch.load("backend/data/train_graphs.pt")
    val_graphs = torch.load("backend/data/val_graphs.pt")
    
    print(f"Training samples: {len(train_graphs)}")
    print(f"Validation samples: {len(val_graphs)}")
    
    # Train VAE
    vae_model = train_vae(train_graphs, val_graphs, epochs=50)
    
    # Train RL
    rl_model = train_rl(vae_model, total_timesteps=50000)
    
    print("Training completed!")

if __name__ == "__main__":
    main()
```
